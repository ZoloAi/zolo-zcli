zCLI + BifrostClient.js - 14 lines

// Frontend with BifrostClient
function addUser() {
  document.getElementById('content').innerHTML = 
    '<h2>➕ Add User</h2>';
  
  client.renderForm([
    { name: 'name', label: 'Name', 
      type: 'text', required: true },
    { name: 'email', label: 'Email', 
      type: 'email', required: true }
  ], '#content', async (data) => {
    await client.send({ 
      zKey: '^Add User', 
      action: 'add_user', data 
    });
    client.renderMessage('✅ User added!', 
      'success', '#content');
  });
}
